def prepare_df(df):
    df = df.set_index(keys=["ID", "Year"])
    df = df.stack()
    df = df.reset_index()
    df = df.rename(columns = {"level_2" : "Month" , 0 : "Temp"})
    df["Month"] = df["Month"].str[5:].astype(int)
    df["Temp"] = df["Temp"] / 100

    return(df)

Part 2: Write a Query Function

def query_climate_database(country, year_begin, year_end, month):

    """
    Creates a query that gives a data frame with station names, latitude, longitude,
    country of origin, year, month, temperature between year_begin and year_end, and
    the actual temperature.
    """

    #connect to database
    conn = sqlite3.connect('data.db')

    cmd = \
    f'''
    SELECT S.name, S.latitude, S.longitude, N.name, T.year, T.Month, T.temp
    FROM temperatures T
    LEFT JOIN stations S ON T.id = S.id
    LEFT JOIN countries N on SUBSTRING(T.id, 1, 2) = N."FIPS 10-4"
    WHERE T.year >= {year_begin} AND T.year <= {year_end} AND T.month == {month} AND N.name = "{country}"
    '''

    #creates dataframe using our above specifications
    df = pd.read_sql_query(cmd, conn)
    conn.close()

    return df

Part 3. Write a Geographic Scatter Function for Yearly Temperature Increases

from sklearn.linear_model import LinearRegression
def coefficient(data):

 """
 Gets coefficients of Year by use of Linear Regression model which are used to estimate
 the annual temeprature change
 """

 x=data[['Year']]
 y=data['Temp']
 LR = LinearRegression()
 LR.fit(x,y)
 return LR.coef_[0]
def temperature_coefficient_plot(country, year_begin, year_end, month, min_obs, **kwargs):
    """
    Creates a scatter mapbox plot to show the estimates of yearly increase in temperature in a given month
    """

    #generate data
    data = query_climate_database(country, year_begin, year_end, month)
    #collects the number of obersations at each unique stations
    obs_counts = data["NAME"].value_counts()
    count_min_obs = data[data['NAME'].isin(obs_counts[obs_counts>=min_obs].index)]

    #use helper function `coefficient` to get the yearly changes in terms of temp
    LR_data = count_min_obs.groupby("NAME").apply(coefficient)
    LR_data = LR_data.reset_index()
    #merge our data
    merged_data = pd.merge(LR_data,count_min_obs,on=['NAME'])

    #rename data
    merged_data = merged_data.rename(columns = {0:'Estimated Yearly Increase (°C)'})
    months = {1: "January", 2: "February", 3: "March", 4: "April", 5: "May", 6: "June", 7: "July",
    8: "August", 9: "September", 10: "October", 11: "November", 12: "December"}

    fig = px.scatter_mapbox(merged_data,
    lat = "LATITUDE",
    lon = "LONGITUDE",
    hover_name = "NAME",
    color = "Estimated Yearly Increase (°C)",
    color_continuous_midpoint = 0,
    title = f"Estimates of yearly increase in temperature in {months[month]} for \
    stations in {country}, years {year_begin} - {year_end}",
    **kwargs)

    return fig
color_map = px.colors.diverging.RdGy_r
fig = temperature_coefficient_plot('India',1980,2020,1,
    min_obs = 10,
    zoom = 2,
    mapbox_style = 'carto-positron',
    color_continuous_scale=color_map)
fig.show()

Part 4. Create Two More Interesting Figures

def query_climate_database_2(country1,country2, year_begin, year_end, month):
    """
    Creates a query that gives a data frame with two countries and their average monthly
    temperature between year_begin and year_end.
    """

    #connect to database
    conn = sqlite3.connect('data.db')

    cmd = \
    f'''
    SELECT N.name, T.year, T.Month, avg(T.temp) 'AverageTemp'
    FROM temperatures T
    LEFT JOIN countries N on SUBSTRING(T.id, 1, 2) = N."FIPS 10-4"
    WHERE T.year >= {year_begin} AND T.year <= {year_end} AND T.month == {month}
    AND (N.name == "{country1}" OR N.name == "{country2}")
    group by T.year, N.name
    '''

    #creates dataframe using our above specifications
    df = pd.read_sql_query(cmd, conn)
    conn.close()

    return df
def temp_heatmap( country1, country2, year_begin, year_end, month):

    """
    Creates a heatmap plot to show monthly mean temperature from year_begin to year_end of
    a specified month.
    """


    # first we get our data
    data = query_climate_database_2(country1, country2, year_begin, year_end, month)

    months = {1: "January", 2: "February", 3: "March", 4: "April", 5: "May", 6: "June", 7: "July",
    8: "August", 9: "September", 10: "October", 11: "November", 12: "December"}

    # plot the heatmap
    fig = px.density_heatmap(data,
    labels = dict(x = "Month", y = "Year", color = "AverageTemp"),
    x = "Month",
    y = "Year",
    z = "AverageTemp",
    facet_col = "Name",
    nbinsx = 12,
    nbinsy = year_end-year_begin+1,
    title = f"Heatmap showing mean temperatures of {months[month]} of countries \
    between {year_begin} and {year_end}")
    return fig
color_map = px.colors.diverging.RdGy_r

def query_climate_database_3(year_begin, year_end, latitude_begin, latitude_end ):
    """
    Creates a query that gives a data frame with for regions below and above a specified latitude
    and their annual mean temperature in two different years.
    """

    #connect to database
    conn = sqlite3.connect('data.db')

    cmd = \
    f'''
    SELECT S.latitude, N.name, T.year, avg(T.temp) 'Average Temp (°C)', T.month
    FROM temperatures T
    LEFT JOIN stations S ON T.id = S.id
    LEFT JOIN countries N on SUBSTRING(T.id, 1, 2) = N."FIPS 10-4"
    WHERE (T.year = {year_begin} OR T.year = {year_end})
    AND (S.latitude >= {latitude_begin} OR S.latitude <= {latitude_end})
    group by T.year, N.name
    '''

    #creates dataframe using our above specifications
    df = pd.read_sql_query(cmd, conn)
    conn.close()

    return df
def temp_choropleth(year_begin, year_end, latitude_begin, latitude_end):
    """
    Creates a choropleth plot to show mean annual temperature
    in year_begin and year_end of a spesified latitudinal region.
    """


    # first we get our data
    data = query_climate_database_3(year_begin, year_end, latitude_begin, latitude_end)


    # plot the choropleth
    fig = px.choropleth(data,
    locations="Name",
    locationmode="country names",
    color="Average Temp (°C)",
    height=300,
    facet_col='Year',
    title = f"Choropleth plot of countries with latitudes below {latitude_end} °N \
    and above {latitude_begin} °N in {year_begin} and {year_end}")


    fig.update_layout(margin={"r":0,"t":50,"b":0,"l":0})

    return fig
